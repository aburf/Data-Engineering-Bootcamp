services:
    postgres:
        image: postgres:13
        environment:
            POSTGRES_USER: airflow
            POSTGRES_password: airflow
            POSTGRES_DB: airflow
        volumes:
            -postgres-db-volume:/var/lib/postgresql/data
        HEALTHCHECK:
            test: ["CMD", "pg_isready", "-U", "airflow"]
            intervals: 5s
            retries: 5
        restart: always

docker volume create --name dtc_postgres_volume_local -d local

    -e POSTGRES_USER: "root" \
